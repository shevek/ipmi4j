005package org.anarres.ipmi.protocol.packet.ipmi.security;
006
007import java.nio.ByteBuffer;
008import javax.annotation.Nonnull;
009import org.anarres.ipmi.protocol.packet.common.AbstractWireable;
010import org.anarres.ipmi.protocol.packet.common.Code;
011
012/**
013 *
014 * @author shevek
015 */
016public class IpmiAlgorithmUtils {
017
018    public static byte getWireLength() {
019        return 8;
020    }
021
022    public static void toWireUnchecked(@Nonnull ByteBuffer buffer, @Nonnull IpmiAlgorithm algorithm) {
023        buffer.put(algorithm.getPayloadType());
024        buffer.putChar((char) 0);      // reserved
025        buffer.put(getWireLength());
026        buffer.put(algorithm.getCode());
027        buffer.put((byte) 0);    // reserved
028        buffer.putChar((char) 0);      // reserved
029    }
030
031    @Nonnull
032    public static <T extends Enum<T> & IpmiAlgorithm> T fromWireUnchecked(@Nonnull ByteBuffer buffer, @Nonnull Class<T> algorithmType) {
033        byte algorithmTypeByte = algorithmType.getEnumConstants()[0].getPayloadType();
034        AbstractWireable.assertWireByte(buffer, algorithmTypeByte, "Algorithm type byte");
035        AbstractWireable.assertWireBytesZero(buffer, 2);
036        AbstractWireable.assertWireByte(buffer, getWireLength(), "Algorithm wire length");
037        T algorithm = Code.fromByte(algorithmType, buffer.get());
038        AbstractWireable.assertWireBytesZero(buffer, 2);
039        buffer.get();   // This is set to 8 in the packet trace I grabbed.
040        return algorithm;
041    }
042}
