005package org.anarres.ipmi.protocol.packet.asf;
006
007import com.google.common.base.Charsets;
008import com.google.common.base.Preconditions;
009import com.google.common.primitives.UnsignedBytes;
010import java.nio.ByteBuffer;
011import java.util.Arrays;
012import javax.annotation.CheckForNull;
013import javax.annotation.Nonnull;
014import org.anarres.ipmi.protocol.client.visitor.IpmiClientAsfMessageHandler;
015import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
016import org.anarres.ipmi.protocol.packet.common.Bits;
017import org.anarres.ipmi.protocol.packet.common.Code;
018
019/**
020 * RAKPMessage1.
021 * 
022 * http://www.dmtf.org/sites/default/files/standards/documents/DSP0136.pdf
023 * http://www.dmtf.org/standards/asf
024 * Section 3.2.4.13 page 42.
025 *
026 * @author shevek
027 */
028public class AsfRAKPMessage1Data extends AbstractAsfData {
029
030    /** Section 3.2.4.13 page 42. */
031    public static enum UserRole implements Bits.Wrapper, Code.Wrapper {
032
033        OPERATOR(new Bits(0, 0b1111, 0b0000)),
034        ADMINISTRATOR(new Bits(0, 0b1111, 0b0001));
035        private final Bits bits;
036
037        private UserRole(@Nonnull Bits bits) {
038            this.bits = bits;
039        }
040
041        @Override
042        public Bits getBits() {
043            return bits;
044        }
045
046        @Override
047        public byte getCode() {
048            return (byte) getBits().getByteValue();
049        }
050    }
051    private int clientSessionId;
052    private byte[] consoleRandom;   // length = 16
053    private UserRole consoleUserRole = UserRole.OPERATOR;
054    private String consoleUserName;
055
056    public int getClientSessionId() {
057        return clientSessionId;
058    }
059
060    @Nonnull
061    public AsfRAKPMessage1Data withClientSessionId(int clientSessionId) {
062        this.clientSessionId = clientSessionId;
063        return this;
064    }
065
066    @Nonnull
067    public byte[] getConsoleRandom() {
068        return consoleRandom;
069    }
070
071    @Nonnull
072    public AsfRAKPMessage1Data withConsoleRandom(@Nonnull byte[] consoleRandom) {
073        Preconditions.checkNotNull(consoleRandom, "Random must be non-null.");
074        Preconditions.checkArgument(consoleRandom.length == 16, "Random length must be 16.");
075        this.consoleRandom = consoleRandom;
076        return this;
077    }
078
079    @Nonnull
080    public UserRole getConsoleUserRole() {
081        return consoleUserRole;
082    }
083
084    @Nonnull
085    public AsfRAKPMessage1Data withConsoleUserRole(UserRole consoleUserRole) {
086        this.consoleUserRole = consoleUserRole;
087        return this;
088    }
089
090    @CheckForNull
091    public String getConsoleUserName() {
092        return consoleUserName;
093    }
094
095    @Nonnull
096    public AsfRAKPMessage1Data withConsoleUserName(String consoleUserName) {
097        this.consoleUserName = consoleUserName;
098        return this;
099    }
100
101    @Override
102    public AsfRmcpMessageType getMessageType() {
103        return AsfRmcpMessageType.RAKPMessage1;
104    }
105
106    @Override
107    public void apply(IpmiClientAsfMessageHandler handler, IpmiHandlerContext context) {
108        handler.handleAsfRAKPMessage1Data(context, this);
109    }
110
111    @Override
112    protected int getDataWireLength() {
113        return 40;  // I bet this is variable in reality.
114    }
115
116    @Override
117    protected void toWireData(ByteBuffer buffer) {
118        buffer.putInt(getClientSessionId());
119        buffer.put(getConsoleRandom());
120        buffer.put(getConsoleUserRole().getCode());
121        buffer.putShort((short) 0); // Reserved, 0
122
123        String userName = getConsoleUserName();
124        if (userName != null) {
125            byte[] userNameBytes = userName.getBytes(Charsets.US_ASCII);
126            buffer.put(UnsignedBytes.checkedCast(userNameBytes.length));
127            buffer.put(Arrays.copyOf(userNameBytes, 16));   // XXX Wrong: It should be variable?
128        } else {
129            buffer.put(new byte[17]);
130        }
131    }
132
133    @Override
134    protected void fromWireData(ByteBuffer buffer) {
135        withClientSessionId(buffer.getInt());
136        withConsoleRandom(readBytes(buffer, 16));
137        withConsoleUserRole(Code.fromBuffer(UserRole.class, buffer));
138        assertWireBytesZero(buffer, 2);
139
140        int userNameLength = UnsignedBytes.toInt(buffer.get());
141        if (userNameLength > 0) {
142            byte[] userNameBytes = readBytes(buffer, userNameLength);
143            withConsoleUserName(new String(userNameBytes, Charsets.US_ASCII));
144        }
145    }
146}
