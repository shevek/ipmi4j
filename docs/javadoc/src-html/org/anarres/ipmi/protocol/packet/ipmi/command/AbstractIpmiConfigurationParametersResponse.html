005package org.anarres.ipmi.protocol.packet.ipmi.command;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009
010/**
011 *
012 * @author shevek
013 */
014public abstract class AbstractIpmiConfigurationParametersResponse extends AbstractIpmiResponse {
015
016    private int presentRevision;
017    private int lastCompatibleRevision;
018    private byte[] parameterData;
019
020    @Override
021    protected int getResponseDataWireLength() {
022        return 2 + parameterData.length;
023    }
024
025    @Override
026    protected void toWireData(ByteBuffer buffer) {
027        if (toWireCompletionCode(buffer))
028            return;
029        int tmp = (presentRevision & 0xF) << 4 | lastCompatibleRevision & 0xF;
030        buffer.put((byte) tmp);
031        buffer.put(parameterData);
032    }
033
034    @Override
035    protected void fromWireData(ByteBuffer buffer) {
036        if (fromWireCompletionCode(buffer))
037            return;
038        int tmp = UnsignedBytes.toInt(buffer.get());
039        presentRevision = (tmp >> 4) & 0xF;
040        lastCompatibleRevision = (tmp >> 4) & 0xF;
041        parameterData = readBytes(buffer, buffer.remaining());
042    }
043
044    @Override
045    public void toStringBuilder(StringBuilder buf, int depth) {
046        super.toStringBuilder(buf, depth);
047        appendValue(buf, depth, "PresentRevision", presentRevision);
048        appendValue(buf, depth, "LastCompatibleRevision", lastCompatibleRevision);
049        appendValue(buf, depth, "ParameterData", toHexString(parameterData));
050    }
051}
