005package org.anarres.ipmi.protocol.packet.ipmi.command.messaging;
006
007import com.google.common.primitives.UnsignedBytes;
008import java.nio.ByteBuffer;
009import javax.annotation.Nonnull;
010import org.anarres.ipmi.protocol.client.session.IpmiSession;
011import org.anarres.ipmi.protocol.client.visitor.IpmiClientIpmiCommandHandler;
012import org.anarres.ipmi.protocol.client.visitor.IpmiHandlerContext;
013import org.anarres.ipmi.protocol.packet.ipmi.IpmiCommandName;
014import org.anarres.ipmi.protocol.packet.ipmi.command.AbstractIpmiRequest;
015
016/**
017 * [IPMI2] Section 22.19, table 22-24, page 297.
018 *
019 * @author shevek
020 */
021public class CloseSessionRequest extends AbstractIpmiRequest {
022
023    public int sessionId;
024    public byte sessionHandle;
025
026    public CloseSessionRequest() {
027    }
028
029    public CloseSessionRequest(int sessionId) {
030        this.sessionId = sessionId;
031    }
032
033    public CloseSessionRequest(@Nonnull IpmiSession session) {
034        this(session.getConsoleSessionId());
035    }
036
037    public CloseSessionRequest withSessionId(int sessionId) {
038        this.sessionId = sessionId;
039        return this;
040    }
041
042    public CloseSessionRequest withSessionHandle(byte sessionHandle) {
043        this.sessionHandle = sessionHandle;
044        return this;
045    }
046
047    @Override
048    public IpmiCommandName getCommandName() {
049        return IpmiCommandName.CloseSession;
050    }
051
052    @Override
053    public void apply(IpmiClientIpmiCommandHandler handler, IpmiHandlerContext context) {
054        handler.handleCloseSessionRequest(context, this);
055    }
056
057    @Override
058    public int getDataWireLength() {
059        return (sessionId == 0) ? 5 : 4;
060    }
061
062    @Override
063    protected void toWireData(ByteBuffer buffer) {
064        toWireIntLE(buffer, sessionId);
065        if (sessionId == 0)
066            buffer.put(sessionHandle);
067    }
068
069    @Override
070    protected void fromWireData(ByteBuffer buffer) {
071        sessionId = fromWireIntLE(buffer);
072        if (sessionId == 0)
073            sessionHandle = buffer.get();
074    }
075
076    @Override
077    public void toStringBuilder(StringBuilder buf, int depth) {
078        super.toStringBuilder(buf, depth);
079        appendValue(buf, depth, "SessionId", "0x" + Integer.toHexString(sessionId));
080        appendValue(buf, depth, "SessionHandle", "0x" + UnsignedBytes.toString(sessionHandle, 16));
081    }
082}
