005package org.anarres.ipmi.protocol.packet.asf;
006
007import com.google.common.collect.Iterables;
008import java.nio.ByteBuffer;
009import java.util.Arrays;
010import java.util.EnumSet;
011import java.util.Set;
012import javax.annotation.Nonnegative;
013import javax.annotation.Nonnull;
014import org.anarres.ipmi.protocol.packet.common.Bits;
015
016/**
017 * PowerUp, Reset and PowerCycleReset.
018 * 
019 * http://www.dmtf.org/sites/default/files/standards/documents/DSP0136.pdf
020 * http://www.dmtf.org/standards/asf
021 * Section 3.2.4.1 pages 33-35.
022 * 
023 * @author shevek
024 */
025public abstract class AbstractAsfBootData extends AbstractAsfData {
026
027    /** [ASF2] Section 3.2.4.1, page 33. */
028    public static class SpecialCommand {
029
030        public static final byte NOP = 0x00;
031        /** Parameter: 0. */
032        public static final byte FORCE_BOOT_PXE = 0x01;
033        /** Parameter: 0 = boot default; 1..n = boot drive n. */
034        public static final byte FORCE_BOOT_DRIVE = 0x02;
035        /** Parameter: 0 = boot default; 1..n = boot drive n. */
036        public static final byte FORCE_BOOT_DRIVE_SAFE = 0x03;
037        /** Parameter: 0. */
038        public static final byte FORCE_BOOT_DIAGNOSTIC = 0x04;
039        /** Parameter: 0 = boot default; 1..n = boot drive n. */
040        public static final byte FORCE_BOOT_OPTICAL = 0x05;
041    }
042
043    /** [ASF2] Section 3.2.4.1, page 34. */
044    public static enum BootOption implements Bits.Wrapper {
045
046        LOCK_POWER_BUTTON(0, 1),
047        LOCK_RESET_BUTTON(0, 2),
048        LOCK_KEYBOARD(0, 5),
049        LOCK_SLEEP_BUTTON(0, 6),
050        USER_PASSWORD_BYPASS(1, 3),
051        FORCE_PROGRESS_EVENTS(1, 4),
052        FIRMWARE_VERBOSITY_DEFAULT(new Bits(1, 0b11 << 5, 0b00 << 5)),
053        FIRMWARE_VERBOSITY_QUIET(new Bits(1, 0b11 << 5, 0b01 << 5)),
054        FIRMWARE_VERBOSITY_VERBOSE(new Bits(1, 0b11 << 5, 0b10 << 5)),
055        FIRMWARE_VERBOSITY_SCREEN_BLANK(new Bits(1, 0b11 << 5, 0b11 << 5)),
056        CONFIGURATION_DATA_RESET(1, 7);
057        private final Bits bits;
058
059        private BootOption(Bits bits) {
060            this.bits = bits;
061        }
062
063        private BootOption(@Nonnegative int byteIndex, @Nonnegative int bitIndex) {
064            this(Bits.forBitIndex(byteIndex, bitIndex));
065        }
066
067        @Override
068        public Bits getBits() {
069            return bits;
070        }
071    }
072    private byte command = SpecialCommand.NOP;
073    private char commandParameter = 0;
074    private final Set<BootOption> options = EnumSet.noneOf(BootOption.class);
075
076    public byte getSpecialCommand() {
077        return command;
078    }
079
080    @Nonnull
081    public AbstractAsfBootData withSpecialCommand(byte command) {
082        this.command = command;
083        return this;
084    }
085
086    public char getSpecialCommandParameter() {
087        return commandParameter;
088    }
089
090    @Nonnull
091    public AbstractAsfBootData withSpecialCommandParameter(char commandParameter) {
092        this.commandParameter = commandParameter;
093        return this;
094    }
095
096    @Nonnull
097    public AbstractAsfBootData withBootOptions(Iterable<? extends BootOption> options) {
098        Iterables.addAll(this.options, options);
099        return this;
100    }
101
102    @Nonnull
103    public AbstractAsfBootData withBootOptions(BootOption... options) {
104        return withBootOptions(Arrays.asList(options));
105    }
106
107    @Nonnull
108    public Set<? extends BootOption> getBootOptions() {
109        return options;
110    }
111
112    @Override
113    public int getDataWireLength() {
114        return 11;
115    }
116
117    @Override
118    public void toWireData(ByteBuffer buffer) {
119        buffer.putInt(IANA_ENTERPRISE_NUMBER.getNumber());
120        buffer.put(getSpecialCommand());
121        buffer.putChar(getSpecialCommandParameter());
122        buffer.put(Bits.toBytes(2, getBootOptions()));
123        buffer.put(new byte[2]);    // TODO: OEM parameters.
124    }
125
126    @Override
127    protected void fromWireData(ByteBuffer buffer) {
128        assertWireInt(buffer, IANA_ENTERPRISE_NUMBER.getNumber(), "IANA enterprise number");
129        withSpecialCommand(buffer.get());
130        withSpecialCommandParameter(buffer.getChar());
131        withBootOptions(Bits.fromBuffer(BootOption.class, buffer, 2));
132        assertWireByte(buffer, (byte) 0, "OEM parameter 0");
133        assertWireByte(buffer, (byte) 0, "OEM parameter 1");
134    }
135}
