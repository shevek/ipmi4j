144    private IpmiChannelNumber channelNumber;

164
165    @Override
166    protected int getResponseDataWireLength() {
167        return 10;
168    }
169
170    @Override
171    protected void toWireData(ByteBuffer buffer) {
172        if (toWireCompletionCode(buffer))
173            return;
174        buffer.put(channelNumber.getCode());
175        buffer.put(channelMedium.getCode());
176        buffer.put(channelProtocol.getCode());
177
178        int tmp = channelSessionSupport.getCode() << 6 | channelSessionCount & 0x3F;
179        buffer.put((byte) tmp);
180
181        toWireOemIanaLE3(buffer, oemEnterpriseNumber);
182
183        if (IpmiChannelNumber.CF.equals(channelNumber)) {
184            buffer.put(smsInterruptType.getCode());
185            buffer.put(eventMessageBufferInterruptType.getCode());
186        } else if (oemEnterpriseNumber != IanaEnterpriseNumber.Intelligent_Platform_Management_Interface_forum.getNumber()) {
187            buffer.put(oem0);
188            buffer.put(oem1);
189        } else {
190            buffer.putChar((char) 0);
191        }
192    }
193
194    @Override
195    protected void fromWireData(ByteBuffer buffer) {
196        if (fromWireCompletionCode(buffer))
197            return;
198        channelNumber = Code.fromBuffer(IpmiChannelNumber.class, buffer);
199        channelMedium = Code.fromBuffer(IpmiChannelMedium.class, buffer);
200        channelProtocol = Code.fromBuffer(IpmiChannelProtocol.class, buffer);
201
202        int tmp = UnsignedBytes.toInt(buffer.get());
203        channelSessionSupport = Code.fromInt(ChannelSessionSupport.class, tmp >> 6);
204        channelSessionCount = tmp & 0x3F;
205
206        oemEnterpriseNumber = fromWireOemIanaLE3(buffer);
207        if (IpmiChannelNumber.CF.equals(channelNumber)) {
208            smsInterruptType = Code.fromBuffer(ChannelInterruptType.class, buffer);
209            eventMessageBufferInterruptType = Code.fromBuffer(ChannelInterruptType.class, buffer);
210        } else if (oemEnterpriseNumber != IanaEnterpriseNumber.Intelligent_Platform_Management_Interface_forum.getNumber()) {
211            oem0 = buffer.get();
212            oem1 = buffer.get();
213        } else {
214            assertWireBytesZero(buffer, 2);
215        }
216    }

229        appendValue(buf, depth, "OEM Data", toHexString(oem0, oem1));
230    }
