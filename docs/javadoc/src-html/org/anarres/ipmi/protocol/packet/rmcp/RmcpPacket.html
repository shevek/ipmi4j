001package org.anarres.ipmi.protocol.packet.rmcp;
002
003import java.nio.ByteBuffer;
004import javax.annotation.Nonnegative;
005import org.anarres.ipmi.protocol.client.session.IpmiPacketContext;
006
007/**
008 * RMCP Packet.
009 *
010 * [ASF2] Section 3.2.3 page 23.
011 *
012 * Uses UDP port 0x026F.
013 *
014 * @author shevek
015 */
016public class RmcpPacket extends AbstractPacket {
017
018    public static final int PORT = 0x26F;
019
020    /** [IPMI2] Page 134, footnote 1. */
021    private static boolean isPaddingRequired(@Nonnegative int length) {
022        switch (length) {
023            case 56:
024            case 84:
025            case 112:
026            case 128:
027            case 156:
028                return true;
029            default:
030                return false;
031        }
032    }
033
034    @Override
035    public int getWireLength(IpmiPacketContext context) {
036        int length = getRawWireLength(context);
037        // if (isPaddingRequired(length)) length++;
038        return length;
039    }
040
041    @Override
042    protected void toWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
043        toWireRaw(context, buffer);
044        // TODO: Compute this based on buffer position rather than re-walking the packet.
045        // if (isPaddingRequired(getRawWireLength(context))) buffer.put((byte) 0);
046    }
047
048    @Override
049    protected final void fromWireUnchecked(IpmiPacketContext context, ByteBuffer buffer) {
050        fromWireRaw(context, buffer);
051        // There may or may not be one additional byte here.
052    }
053
054    @Override
055    public void toStringBuilder(StringBuilder buf, int depth) {
056        appendHeader(buf, depth, "PacketHeader");
057        appendValue(buf, depth, "RemoteAddress", getRemoteAddress());
058        appendHeader(buf, depth, "RmcpHeader");
059        super.toStringBuilder(buf, depth + 1);  // Header
060        appendChild(buf, depth, "RmcpData", getData());
061    }
062}
